import lynxTypes::*;

/**
 * @brief Shell control module
 * This module generates three clocks (shell, network, user) from the static layer clock
 * Additionaly, it implements a crossbar, spliting the main AXI4 interface into multiple AXI4-Lite interfaces
 * One to access configuration registers of the shell and one per region (vFPGA), which is further split
 * by the dynamic crossbar into AXI4-Lite interfaces for vFPGA config, small and large TLB and user CSRs
 */
module shell_ctrl_cc (
    AXI4.s              axi_main,

    AXI4L.m             axi_cnfg,
    AXI4L.m             axi_ctrl [N_REGIONS],
{% if cnfg.en_avx %}
    AXI4.m              axim_ctrl [N_REGIONS],
{% endif %}
{% if cnfg.en_nvme %}
    // axi_nvme_cnfg sliced from axi_cnfg by nvme_cnfg_axil_split
    AXI4.m              axi_nvme_prp,
    AXI4.m              axi_nvme_sq,
    AXI4.m              axi_nvme_cq,
    {% if cnfg.en_mem %}
    AXI4.m              axi_card_mem,
    {% endif %}
{% endif %}

{% if cnfg.fpga_arch == 'versal' and not cnfg.en_pr %}
    AXI4.s              axi_debug_hub,
{% endif %}

    input  logic        xclk,
    input  logic        xresetn,
    output logic        aclk,
    output logic        aresetn,
    output logic        nclk,
    output logic        nresetn,
    output logic        uclk,
    output logic        uresetn,
    output logic        lckresetn,
    input  logic        sys_reset
);
